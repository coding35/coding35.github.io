A FullAdder is a digital logic gate that adds three binary 1-bit inputs and produces a sum and carry output. The FullAdder outputs two bits that taken together, represent the addition of the three input bits.

HDL

        CHIP FullAdder {
            IN a, b, c;  // 1-bit inputs
            OUT sum,     // Right bit of a + b + c
                carry;   // Left bit of a + b + c
        
            PARTS:
                Xor(a=a, b=b, out=xor1);
                Xor(a=c, b=xor1, out=sum);
                And(a=c, b=xor1, out=and1);
                And(a=a, b=b, out=and2);
                Or(a=and1, b=and2, out=carry);
        }
