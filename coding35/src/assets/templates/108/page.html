The Mux8Way16 gate is a 16-bit digital logic gate that selects one of the 16 inputs based on the value of the select input.

HDL

        /**
        * 8-way 16-bit multiplexor:
        * out = a if sel = 000
        *       b if sel = 001
        *       c if sel = 010
        *       d if sel = 011
        *       e if sel = 100
        *       f if sel = 101
        *       g if sel = 110
        *       h if sel = 111
        */
       CHIP Mux8Way16 {
           IN a[16], b[16], c[16], d[16],
              e[16], f[16], g[16], h[16],
              sel[3];
           OUT out[16];
       
           PARTS:
           Mux4Way16(a=a, b=b, c=c, d=d, sel=sel[0..1], out=ad);
           Mux4Way16(a=e, b=f, c=g, d=h, sel=sel[0..1], out=eh);
           Mux16(a=ad, b=eh, sel=sel[2], out=out);
       }
